{/* E-commerce Checkout — example product model */}
{/* version: 1.0.0 */}
{/* title: Checkout Flow */}

<Feature id="checkout" name="Checkout Flow">
  Comprehensive checkout flow from cart preparation to order confirmation.

  {/* ── Actors ─────────────────────────────────────────────── */}

  <Actor id="guest-shopper" name="Guest Shopper"
    description="Unauthenticated visitor. No saved payment methods or address book."
  />
  <Actor id="member-shopper" name="Member Shopper"
    description="Authenticated user with a verified account, saved addresses, and payment methods on file."
  />
  <Actor id="fraud-system" name="Fraud System"
    description="Internal automated system that performs risk scoring and velocity checks."
  />

  {/* ── Outcomes ───────────────────────────────────────────── */}

  <Outcome id="checkout-conversion" name="Checkout Conversion Rate"
    metric="checkout_conversion_rate"
    baseline="68%"
    target="75%"
    timeframe="30d"
    owner="checkout-pm"
    description="Percentage of users who reach cart and complete a paid order."
  />
  <Outcome id="payment-decline-rate" name="Payment Decline Rate"
    metric="payment_decline_rate"
    baseline="7%"
    target="4%"
    timeframe="30d"
    owner="payments-pm"
    description="Percentage of payment attempts that are declined, including soft declines."
  />

  {/* ── Cart ───────────────────────────────────────────────── */}

  <Section id="cart" name="Shopping Cart">
    Captures line items and shopping constraints.

    <Definition id="cart-item" name="Cart Item" version="1.0.0">
      Canonical structure for one line item in checkout.
      <Field name="productId" type="string" required />
      <Field name="quantity" type="number" required min={1} max={99} />
      <Field name="unitPrice" type="number" required />
      <Field name="currency" type="enum" values="USD,EUR,GBP,KRW" required />
    </Definition>

    <Policy id="max-cart-items" name="Maximum Cart Items"
      trigger="cart.item.add"
      actor="guest-shopper member-shopper"
      rule="A cart may contain at most 50 distinct line items"
      enforcement="must"
    >
      Prevent oversized carts that degrade fulfillment and payment UX.
      <Logic id="max-cart-items-logic" name="Max Cart Item Guard">
        On add-item request, count distinct items; reject with error code CART_LIMIT_EXCEEDED when count exceeds 50.
      </Logic>
    </Policy>

    <Policy id="positive-quantity" name="Positive Quantity"
      trigger="cart.item.update"
      actor="guest-shopper member-shopper"
      rule="Quantity must be at least 1 and at most 99 per line item"
      enforcement="must"
    >
      Keep quantity values within safe operational limits.
      <Logic id="positive-quantity-logic" name="Quantity Bounds Validation">
        Accept quantity only when it is between 1 and 99 inclusive. Return INVALID_QUANTITY if out of range; clamp silently only when triggered by inventory sync.
      </Logic>
    </Policy>

    <Policy id="inventory-hold" name="Inventory Hold on Cart Add"
      trigger="cart.item.add"
      actor="guest-shopper member-shopper"
      rule="Adding an item to cart must create a soft inventory hold that expires after 15 minutes of inactivity"
      enforcement="should"
    >
      Prevents overselling for high-demand products while avoiding permanent lockup of stock from abandoned carts.
      <Logic id="inventory-hold-create" name="Hold Creation Logic">
        On add-to-cart, call inventory service to create a soft hold with a 15-minute TTL. If hold fails due to insufficient stock, return OUT_OF_STOCK and prevent the item from being added.
      </Logic>
      <Logic id="inventory-hold-extend" name="Hold Extension Logic">
        Each cart interaction (quantity change, add item, proceed to checkout) resets the TTL to 15 minutes from the current timestamp. Moving to payment extends the hold to 30 minutes.
      </Logic>
      <Logic id="inventory-hold-release" name="Hold Release Logic">
        On TTL expiry, release the hold and mark the cart item as unconfirmed. On next cart view, re-check availability and notify the user if stock is no longer available.
      </Logic>
    </Policy>

    <Policy id="promo-code-validation" name="Promo Code Validation"
      trigger="cart.promo.apply"
      actor="guest-shopper member-shopper"
      rule="Only one promo code may be applied per cart; the code must be valid, unexpired, and satisfy minimum order requirements"
      enforcement="must"
    >
      Ensures promotional discounts are applied correctly and prevents stacking exploits.
      <Logic id="promo-code-apply" name="Promo Code Application Logic">
        On code submission, validate against promo service: check code exists, is within valid date range, and cart subtotal meets minimumOrderValue. Apply discount as either percentage or fixed amount based on promo type. Return INVALID_PROMO_CODE, PROMO_EXPIRED, or MINIMUM_NOT_MET as appropriate.
      </Logic>
      <Logic id="promo-code-recalculate" name="Promo Code Recalculation Logic">
        When cart contents change after promo application, re-evaluate eligibility. If cart no longer meets minimum, remove promo and notify user.
      </Logic>
    </Policy>

    <Constraint id="cart-currency-consistency" name="Cart Currency Consistency"
      condition="All line items in a cart must share the same currency"
      actor="guest-shopper member-shopper"
    >
      Prevents mixed-currency carts that break tax and payment calculations.
    </Constraint>

    <Scenario id="sc-cart-at-limit" name="Cart at 50-item capacity"
      given="A cart containing exactly 50 distinct line items"
      when="The user attempts to add a 51st distinct item"
      then="The system rejects the request and returns error code CART_LIMIT_EXCEEDED without modifying the cart"
      policy="max-cart-items"
      actor="guest-shopper"
    />
    <Scenario id="sc-quantity-out-of-range" name="Quantity exceeds maximum"
      given="A cart with one item at quantity 99"
      when="The user attempts to set the quantity to 100"
      then="The system rejects the update and returns INVALID_QUANTITY"
      policy="positive-quantity"
      actor="member-shopper"
    />
  </Section>

  {/* ── Payment ─────────────────────────────────────────────── */}

  <Section id="payment" name="Payment Processing">
    Manages payment method capture and gateway constraints.

    <Definition id="payment-method" name="Payment Method" version="1.0.0">
      Normalized payment representation for authorization and capture.
      <Field name="type" type="enum" values="credit_card,debit_card,paypal,apple_pay,google_pay" required />
      <Field name="token" type="string" required />
      <Field name="last4" type="string" required={false} pattern="^\d{4}$" />
    </Definition>

    <Policy id="pci-compliance" name="PCI Compliance"
      trigger="payment.token.store"
      rule="Raw card numbers must never be stored; only tokenized references are permitted"
      enforcement="must"
    >
      Enforce PCI-safe handling of payment credentials.
      <Logic id="pci-compliance-logic" name="Tokenization Enforcement">
        Drop PAN fields and persist only provider-issued tokens. Log a security event if raw card data is detected in any request payload.
      </Logic>
    </Policy>

    <Policy id="payment-retry" name="Payment Retry Policy"
      trigger="payment.authorize.fail"
      actor="guest-shopper member-shopper"
      rule="Failed payment attempts should be retried up to 2 times with progressive gateway fallback before surfacing an error to the user"
      enforcement="should"
    >
      Reduces false declines caused by transient gateway issues and improves conversion rate.
      <Logic id="payment-retry-same-gateway" name="Same-Gateway Retry Logic">
        On soft decline (network timeout, 5xx from gateway), wait 2 seconds and retry on the same gateway. If the second attempt also fails with a soft decline, proceed to fallback.
      </Logic>
      <Logic id="payment-retry-fallback" name="Gateway Fallback Logic">
        If primary gateway fails twice, route the third attempt through the secondary gateway. If all three attempts fail, present a user-friendly error. Never retry on hard declines.
      </Logic>
    </Policy>

    <Policy id="payment-idempotency" name="Payment Idempotency"
      trigger="payment.authorize"
      rule="Each payment attempt must carry a unique idempotency key to prevent duplicate charges"
      enforcement="must"
    >
      Prevents double-charging caused by network retries or frontend re-submissions.
      <Logic id="payment-idempotency-logic" name="Idempotency Key Logic">
        Generate a UUID-based idempotency key on checkout initiation. Attach the key to every authorization request. The gateway must reject duplicate keys within a 24-hour window.
      </Logic>
    </Policy>

    <Policy id="fraud-screening" name="Fraud Screening"
      trigger="payment.authorize"
      actor="fraud-system"
      rule="All transactions above $500 or flagged by risk scoring must undergo additional verification before capture"
      enforcement="must"
    >
      Reduces chargeback rate and protects against fraudulent high-value orders.
      <Logic id="fraud-risk-scoring" name="Risk Score Evaluation">
        Before authorization, pass order details to fraud scoring service. Orders with risk score above 80 are auto-declined. Scores between 60-80 trigger step-up verification (3DS or email OTP).
      </Logic>
      <Logic id="fraud-velocity-check" name="Velocity Check Logic">
        Flag accounts that attempt more than 3 failed payments in 10 minutes or more than 5 orders in 24 hours. Temporarily lock checkout for the account.
      </Logic>
    </Policy>

    <Constraint id="payment-timeout" name="Payment Timeout"
      condition="Payment gateway response must arrive within 30 seconds"
      actor="guest-shopper member-shopper"
    >
      Protect user flow from long-running authorization calls.
    </Constraint>

    <Constraint id="currency-consistency" name="Currency Consistency"
      condition="The currency used in payment authorization must match the currency displayed to the user at cart time"
    >
      Prevents exchange rate discrepancies between displayed price and charged amount.
    </Constraint>

    <Scenario id="sc-payment-soft-decline" name="Soft decline triggers retry"
      given="A valid cart and payment method are submitted for authorization"
      when="The primary payment gateway returns a 503 timeout"
      then="The system waits 2 seconds and retries on the same gateway before falling back"
      policy="payment-retry"
      actor="member-shopper"
    />
    <Scenario id="sc-fraud-high-risk" name="High-risk transaction blocked"
      given="An order totalling $800 from a new account with a high-entropy device fingerprint"
      when="The fraud scoring service returns a risk score of 85"
      then="The order is auto-declined with reason FRAUD_RISK_TOO_HIGH and no charge is made"
      policy="fraud-screening"
      actor="fraud-system"
    />
  </Section>

  {/* ── Shipping ─────────────────────────────────────────────── */}

  <Section id="shipping" name="Shipping">
    Shipping destination and fulfillment option data.

    <Definition id="shipping-address" name="Shipping Address" version="1.0.0">
      Address schema for delivery and tax calculation.
      <Field name="street" type="string" required />
      <Field name="city" type="string" required />
      <Field name="state" type="string" required={false} />
      <Field name="postalCode" type="string" required />
      <Field name="country" type="string" required />
    </Definition>

    <Definition id="shipping-option" name="Shipping Option" version="1.0.0">
      Available shipping methods offered at checkout.
      <Field name="carrier" type="string" required />
      <Field name="speed" type="enum" values="standard,express,overnight" required />
      <Field name="price" type="number" required min={0} />
      <Field name="estimatedDays" type="number" required min={1} />
    </Definition>

    <Policy id="address-validation" name="Address Validation"
      trigger="shipping.address.submit"
      actor="guest-shopper member-shopper"
      rule="Shipping addresses must be validated against a postal service API before order placement"
      enforcement="must"
    >
      Reduces failed deliveries and return costs from invalid or incomplete addresses.
      <Logic id="address-validation-logic" name="Address Verification Logic">
        On address submission, call postal validation API. If undeliverable, show an error. If the API suggests corrections, present suggestions and require explicit user confirmation.
      </Logic>
    </Policy>

    <Policy id="restricted-regions" name="Restricted Region Handling"
      trigger="shipping.address.submit"
      actor="guest-shopper member-shopper"
      rule="Orders shipping to sanctioned countries or regions with import restrictions must be blocked at checkout"
      enforcement="must"
    >
      Ensures trade compliance and prevents orders that cannot be legally fulfilled.
      <Logic id="restricted-regions-logic" name="Region Restriction Logic">
        Maintain a configurable blocklist of country and region codes. On address entry, check against the blocklist. If restricted, display a message without disclosing specific compliance details. Log blocked attempts for compliance audit.
      </Logic>
    </Policy>

    <Scenario id="sc-invalid-address" name="Undeliverable address rejected"
      given="A guest shopper has entered a non-existent postal code"
      when="The user clicks 'Continue to Payment'"
      then="The system calls the postal API, receives UNDELIVERABLE, and displays an address correction prompt"
      policy="address-validation"
      actor="guest-shopper"
    />
  </Section>

  {/* ── Order Confirmation ───────────────────────────────────── */}

  <Section id="order-confirmation" name="Order Confirmation">
    Final order record and confirmation policies.

    <Definition id="order" name="Order" version="1.0.0">
      Immutable order snapshot created after successful payment.
      <Field name="orderId" type="string" required pattern="^ORD-[A-Z0-9]{8}$" />
      <Field name="status" type="enum" values="pending,confirmed,shipped,delivered,cancelled" required />
      <Field name="total" type="number" required min={0} />
      <Field name="createdAt" type="datetime" required />
    </Definition>

    <Policy id="confirmation-email" name="Order Confirmation Email"
      trigger="order.created"
      rule="A confirmation email must be sent within 60 seconds of order placement"
      enforcement="must"
    >
      Ensure users get immediate post-purchase confirmation.
      <Logic id="confirmation-email-logic" name="Confirmation Delivery Logic">
        Enqueue confirmation event at order creation. Email service must deliver within 60 seconds. If delivery fails, retry up to 3 times with exponential backoff (5s, 15s, 45s).
      </Logic>
    </Policy>

    <Policy id="cancellation-window" name="Order Cancellation Window"
      trigger="order.cancel.request"
      actor="member-shopper"
      rule="Users may cancel an order within 30 minutes of placement if the order has not entered fulfillment"
      enforcement="must"
    >
      Provides a grace period for accidental purchases while respecting fulfillment workflow constraints.
      <Logic id="cancellation-eligibility" name="Cancellation Eligibility Logic">
        On cancel request, check that createdAt is within 30 minutes of current time AND order status is still pending or confirmed. If eligible, transition status to cancelled, initiate refund, and release inventory holds.
      </Logic>
      <Logic id="cancellation-refund" name="Cancellation Refund Logic">
        Initiate a full refund through the original payment method. If refund fails, create a store credit for the full amount and notify the user.
      </Logic>
    </Policy>

    <Scenario id="sc-cancel-in-window" name="Order cancelled within grace period"
      given="A confirmed order placed 20 minutes ago with status 'confirmed'"
      when="The member clicks 'Cancel Order'"
      then="The order transitions to 'cancelled', a full refund is initiated, and inventory holds are released"
      policy="cancellation-window"
      actor="member-shopper"
    />
    <Scenario id="sc-cancel-after-window" name="Cancel attempt after 30-minute window"
      given="A confirmed order placed 45 minutes ago with status 'confirmed'"
      when="The member attempts to cancel via the order detail page"
      then="The system rejects the cancellation with CANCEL_WINDOW_EXPIRED and displays a customer support contact"
      policy="cancellation-window"
      actor="member-shopper"
    />
  </Section>

  <Link id="shipping-to-confirmation" from="shipping" to="order-confirmation" relationship="depends-on"

  />
</Feature>
